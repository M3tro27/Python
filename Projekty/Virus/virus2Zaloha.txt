"""Proces  zašifrování"""
def encrypt(filename, key):  
   
    f = Fernet(key)
    with open(filename, "rb") as file:
    # precteme vsechna data
        file_data = file.read()

        encrypted_data = f.encrypt(file_data)
    #vytvorríme novy zasifrovany soubor
    with open(filename, "wb") as file:
        file.write(encrypted_data)
